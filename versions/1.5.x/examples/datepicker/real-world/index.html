DatePicker made from markup:

<div id="datePicker">
  <div id="srcNode">
    <select id="yearNode">
      <option></option>
    </select>
    <select id="dayNode">
      <option></option>
    </select>
    <select id="monthNode">
      <option></option>
    </select>
    <button id="trigger">Calendar Trigger</button>
  </div>
</div>
AUI().use(
  'aui-datepicker',
  function(A) {
    new A.DatePickerSelect(
      {
        boundingBox: '#datePicker',
        dayNode: '#dayNode',
        dayNodeName: 'daySelect',
        monthNode: '#monthNode',
        monthNodeName: 'monthSelect',
        render: true,
        srcNode: '#srcNode',
        trigger: '#trigger',
        yearNode: '#yearNode',
        yearNodeName: 'yearSelect',
        yearRange: [1980, 2013]
      }
    );
  }
);
