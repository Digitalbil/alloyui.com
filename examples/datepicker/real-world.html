DatePicker - From Markup

DatePicker made from markup:

<div id="datePicker">
  <div id="srcNode">
    <select id="yearNode">
      <option></option>
    </select>
    <select id="dayNode">
      <option></option>
    </select>
    <select id="monthNode">
      <option></option>
    </select>
    <button id="trigger">Celendar Trigger</button>
  </div>
</div>
AUI().use('aui-datepicker', function(A) {

  new A.DatePickerSelect({
    boundingBox: '#datePicker',
    srcNode: '#srcNode',
    trigger: '#trigger',
    yearNode: '#yearNode',
    monthNode: '#monthNode',
    dayNode: '#dayNode',
    yearNodeName: 'yearSelect',
    monthNodeName: 'monthSelect',
    dayNodeName: 'daySelect',
    yearRange: [ 1980, 2013 ],
    render: true
  });

});
