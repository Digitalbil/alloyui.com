AUI().ready('aui-tooltip', function(A) {

  new A.Tooltip({
    trigger: '#myTooltip',
    bodyContent: '<img src="http://liferay.github.com/alloyui.com/tooltip/img/cream.jpg" />',
    showArrow: false
  }).render();

});
