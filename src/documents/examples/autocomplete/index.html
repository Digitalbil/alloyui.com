--- title: Auto Complete description: AutoComplete allows users to select from a pre-defined list thumbnail: http://alloyui.com/images/demo_images/autocomplete.png layout: single-example --- 
AUI().use('aui-autocomplete', function(A) {

  var continents = ['America', 'Europe', 'Asia', 'Africa', 'Oceania', 'Antarctica'];

  var autoComplete = new A.AutoComplete({
    dataSource: continents,
    contentBox: '#myAutoComplete'
  });

  autoComplete.render();

});
